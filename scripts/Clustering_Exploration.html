ccas <- cascadeKM(decostand(ord.mat, method = "total"), 2, 10)
plot(ccas, sortq=TRUE)

parts = bind_cols(MDS_xy, cluster = ccas$partition[,2])
parts$cluster[parts$sp.trt == "Natives" & parts$year == 2008] = 2
